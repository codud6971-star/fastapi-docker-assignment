# 1. 파이썬 공식 경량화 이미지 가져오기
FROM python:3.10-slim

# 2. 컨테이너 내부의 작업 디렉토리를 /app으로 지정
WORKDIR /app

# 3. 라이브러리 설치 파일만 먼저 복사 (컨테이너 빌드 속도 최적화용)
COPY requirements.txt .

# 4. 파이썬 라이브러리 설치 (캐시를 남기지 않아 용량 최적화)
RUN pip install --no-cache-dir -r requirements.txt

# 5. 현재 내 폴더의 모든 소스 코드(.py, .json 등)를 컨테이너 안으로 복사
COPY . .

# 6. 컨테이너가 켜질 때 자동으로 구동할 명령어 설정
# 외부(EC2) 접속을 허용하기 위해 host는 반드시 0.0.0.0으로 고정해야 합니다.
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
